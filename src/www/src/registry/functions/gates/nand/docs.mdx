## Usage
```javascript
"use js-executor"
import nand from "@/helpers/nand";

console.log(nand(true, true));
// Expected Output: false

console.log(nand(true, false));
// Expected Output: true

console.log(nand());
// Expected Output: false

console.log(nand(1, "lazykit"));
// Expected Output: false
```
