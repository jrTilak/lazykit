import { getCollection } from "astro:content";
import { Card, CardGrid, LinkCard } from "@astrojs/starlight/components";

export const entries = await getCollection("docs");
export const counts = {
  functions: entries.filter(({ data }) => data.kind === "function").length,
  hooks: entries.filter(({ data }) => data.kind === "hook").length,
};
