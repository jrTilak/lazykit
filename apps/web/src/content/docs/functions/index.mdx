import { getCollection } from "astro:content";
import { CardGrid, LinkCard } from "@astrojs/starlight/components";

export const categoryOrder = [
  "Arrays",
  "Comparison",
  "Functional",
  "Logic",
  "Objects",
  "Web APIs",
];

export const functions = (await getCollection("docs"))
  .filter(({ data }) => data.kind === "function" && data.category)
  .sort((a, b) => a.data.title.localeCompare(b.data.title));

export const categories = categoryOrder
  .map((category) => ({
    category,
    entries: functions.filter((entry) => entry.data.category === category),
  }))
  .filter(({ entries }) => entries.length > 0);

LazyKit currently includes **{functions.length} TypeScript functions**. The files
live in one flat documentation collection; the categories below come entirely
from each page's frontmatter.

{
  categories.map(({ category, entries }) => (
    <section>
      <h2>{category}</h2>
      <CardGrid>
        {entries.map((entry) => (
          <LinkCard
            title={entry.data.title}
            description={entry.data.description}
            href={`/${entry.id.replace(/\/index$/, "")}/`}
          />
        ))}
      </CardGrid>
    </section>
  ))
}
